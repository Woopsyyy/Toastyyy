-- Enable UUID extension
create extension if not exists "uuid-ossp";

-- Create tables
create table public.profiles (
  id uuid references auth.users on delete cascade not null primary key,
  username text unique,
  avatar_url text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

create table public.user_dashboards (
  id uuid default uuid_generate_v4() primary key,
  user_id uuid references public.profiles(id) on delete cascade not null,
  title text not null,
  layout jsonb default '[]'::jsonb not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

create table public.graph_presets (
  id uuid default uuid_generate_v4() primary key,
  user_id uuid references public.profiles(id) on delete cascade not null,
  name text not null,
  config jsonb default '{}'::jsonb not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Enable RLS
alter table public.profiles enable row level security;
alter table public.user_dashboards enable row level security;
alter table public.graph_presets enable row level security;

-- Create Policies
-- Profiles: Users can view their own profile and update it. (Others can view username/avatar if we want, but keeping it private for now)
create policy "Users can view own profile" on public.profiles
  for select using ( auth.uid() = id );

create policy "Users can update own profile" on public.profiles
  for update using ( auth.uid() = id );

create policy "Users can insert own profile" on public.profiles
  for insert with check ( auth.uid() = id );

-- Dashboards: Users can fully manage their own dashboards
create policy "Users can view own dashboards" on public.user_dashboards
  for select using ( auth.uid() = user_id );

create policy "Users can insert own dashboards" on public.user_dashboards
  for insert with check ( auth.uid() = user_id );

create policy "Users can update own dashboards" on public.user_dashboards
  for update using ( auth.uid() = user_id );

create policy "Users can delete own dashboards" on public.user_dashboards
  for delete using ( auth.uid() = user_id );

-- Graph Presets: Users can fully manage their own presets
create policy "Users can view own presets" on public.graph_presets
  for select using ( auth.uid() = user_id );

create policy "Users can insert own presets" on public.graph_presets
  for insert with check ( auth.uid() = user_id );

create policy "Users can update own presets" on public.graph_presets
  for update using ( auth.uid() = user_id );

create policy "Users can delete own presets" on public.graph_presets
  for delete using ( auth.uid() = user_id );
